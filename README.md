# config_auto_redmi4x

auto boot when charge
auto adb tcp ip:5555
auto open termux after booting

# How to install

pkg update && pkg upgrade -y

pkg install wget -y

wget https://raw.githubusercontent.com/takumitesla/main/autoconfig.sh

chmod +x autoconfig.sh

su -c "./autoconfig.sh"
